\def\textbox(#1) at (#2) [#3] <#4,#5> title #6{%
  \def\textboxname{#1}%
  \def\textboxpos{#2}%
  \def\textboxalign{#3}%
  \def\textboxncolumns{#4}%
  \setlength{\textboxwidth}{#5}%
  \def\textboxtitle{#6}%
%
  \ifdefstring{\textboxncolumns}{1}{%
    % single column
    \def\atbeginofminipage{%
      \vspace{.5\baselineskip}%
      \par\noindent\ignorespacesandpars}%
    \def\atendofminipage{}%
  }{%
    % multiple columns
    \setlength{\textboxwidth}{\textboxncolumns\textboxwidth}%
    \addtolength{\textboxwidth}{\textboxncolumns\textboxcolumnsep}%
    \addtolength{\textboxwidth}{-\textboxcolumnsep}%
    \def\atbeginofminipage{%
      \setlength{\columnsep}{1em}%
      \begin{multicols}{\textboxncolumns}%
        \par\noindent\ignorespacesandpars}%
    \def\atendofminipage{%
      \end{multicols}}%
  }%
%
  \node[inner sep=0pt,outer sep=0cm] (#1) at (#2) [#3] \bgroup%
    \begin{minipage}{\textboxwidth}%
      \setlength{\parskip}{0pt}%
      \setlength{\parindent}{1em}%
      \par\noindent%
      {% title
        \color{tudcyan}\scshape%
        \centeredhrulefill\hspace{1em}%
        \textboxtitle%
        \hspace{1em}\centeredhrulefill}%
      \atbeginofminipage}%
\def\endtextbox{%
      \atendofminipage%
    \end{minipage}%
  \egroup;%
}

\begin{textbox}(Intermittent Execution Models)
at (summary.north west) [below right] <1,.47\textwidth>
title {Intermittent Execution Models}

\begin{itemize}

\item Checkpointing   

  \begin{enumerate}

    \item \emph{VM Checkpointing}: 
    \begin{enumerate}
    \item It does not scale with RAM size.
    \item \emph{Roll back chkpt}: FRAM interaction problem.
    \item \emph{Push forward chkpt}:  peripherals reinitialization problem (i.e DMA).
    \end{enumerate}

    \item \emph{NVM Checkpointing}: 
    \begin{enumerate}
    \item FRAM consume more energy than RAM 
    \item FRAM is slower 
    \end{enumerate}

  \end{enumerate}

  \centering
      \adjincludegraphics[width=\columnwidth]{./fig/fram_write}

    \vspace{-1cm}
   \textbf{Energy cost of FRAM and RAM write access}

\end{itemize}  
\begin{itemize}

\item Task based

  \begin{enumerate}
  \item static division 
  \item high memory consumption (multi-versioning)
  \item static dependencies  
  \item code modification 
  \end{enumerate}
    
    \vspace{1cm}
    \centering
      \adjincludegraphics[width=0.80\columnwidth]{./fig/dynamic_war}

   \textbf{Selective saving results in dynamic war dependencies. }

\end{itemize}
